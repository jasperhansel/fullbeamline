82 static void
83 plf2ops_c_minmax( PLPointer p, PLINT nx, PLINT ny, PLFLT *zmin, PLFLT *zmax )
84 {
85  int i, j;
86  PLFLT min, max;
87  PLFLT **z = (PLFLT **) p;
88 
89  if ( !isfinite( z[0][0] ) )
90  {
91  max = -HUGE_VAL;
92  min = HUGE_VAL;
93  }
94  else
95  min = max = z[0][0];
96 
97  for ( i = 0; i < nx; i++ )
98  {
99  for ( j = 0; j < ny; j++ )
100  {
101  if ( !isfinite( z[i][j] ) )
102  continue;
103  if ( z[i][j] < min )
104  min = z[i][j];
105  if ( z[i][j] > max )
106  max = z[i][j];
107  }
108  }
109  *zmin = min;
110  *zmax = max;
111 }
